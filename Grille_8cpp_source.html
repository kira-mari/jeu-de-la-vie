100void Grille::mettreAJour(bool parallele) {
101 // Créer une copie de la grille pour les nouveaux états
102 Grille nouvelleGrille(nbLignes, nbColonnes, regle);
103
104 // Décider si la parallélisation vaut la peine
105 bool utiliserParallele = parallele && (nbLignes >= Config::TAILLE_MIN_PARALLELE);
106
107 if (utiliserParallele) {
108 // Version parallélisée optimisée
109 unsigned int nbThreads = std::thread::hardware_concurrency();
110 if (nbThreads == 0) nbThreads = 4;
111 // Limiter le nombre de threads pour éviter l'overhead
112 nbThreads = std::min(nbThreads, static_cast<unsigned int>(nbLignes / 50));
113 nbThreads = std::max(2u, nbThreads); // Au minimum 2 threads
114
115 std::vector<std::thread> threads;
116 threads.reserve(nbThreads);
117 int lignesParThread = nbLignes / nbThreads;
118
119 for (unsigned int t = 0; t < nbThreads; ++t) {
120 int ligneDebut = t * lignesParThread;
121 int ligneFin = (t == nbThreads - 1) ? nbLignes : (ligneDebut + lignesParThread);
122
123 threads.emplace_back([this, ligneDebut, ligneFin, &nouvelleGrille]() {
124 this->mettreAJourLignes(ligneDebut, ligneFin, nouvelleGrille);
125 });
126 }
127
128 for (auto& thread : threads) {
129 thread.join();
130 }
131 } else {
132 // Version séquentielle optimisée
133 for (int i = 0; i < nbLignes; ++i) {
134 for (int j = 0; j < nbColonnes; ++j) {
135 auto nouvelEtat = regle->calculerNouvelEtat(*this, i, j);
136 nouvelleGrille.definirEtatCellule(i, j, std::move(nouvelEtat));
137 }
138 }
139 }
140
141 // Sauvegarder l'état actuel pour vérifier la stabilité (après le calcul pour réutiliser les données)
142 if (!premiereIteration && grilleMemoire) {
143 // Swap au lieu de copier - beaucoup plus rapide
144 std::swap(grilleMemoire->cellules, cellules);
145 } else if (!grilleMemoire) {
146 grilleMemoire = std::make_unique<Grille>(nbLignes, nbColonnes, regle);
147 grilleMemoire->cellules = cellules;
148 }
149
150 // Remplacer l'ancienne grille par la nouvelle (move, pas de copie)
151 cellules = std::move(nouvelleGrille.cellules);
152
